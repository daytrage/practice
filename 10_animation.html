@keyframes 直進回転→直進

    .box10 {
      animation-name: move;
      animation-duration: 2s;
      animation-fill-mode: forwards;　←2s後の状態を維持する
    }
    
    
    /* 名前の付け方：英数字(ハイフン)(アンダーバー) */
    @keyframes move {
      /* 0秒の時 */
      0% {
        /* 変形はしていない */
        transform: none;
      }
      /* 1.6sの時 */
      80% {
        /* 右に80px移動 360度回転*/
        transform: translateX(200px) rotate(360deg);
      }
      /* 2秒の時 */
      100% {
        /* さらに右へ100px移動 360度回転したまま*/
        transform: translateX(300px) rotate(360deg);
      }
    }
